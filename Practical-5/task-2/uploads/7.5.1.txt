using System;

namespace MyApplication
{
        //Base Class
        abstract class Figure
        {
            public abstract int perimetri();
        }

        //Child Class
        class Square : Figure
        {
            public int side1;
            public int side2;
            public Square(int side1, int side2)
            {
                this.side1 = side1;
                this.side2 = side2;
            }
            public override int perimetri()
            {
                return side1 * side2;
            }
        }

        //Child Class
        class Triangle : Figure
        {
            public int side1;
            public int side2;
            public int side3;
        public Triangle(int side1, int side2, int side3)
            {
                this.side1 = side1;
                this.side2 = side2;
                this.side3 = side3;
            }
        public override int perimetri()
        {
            return side1 + side2 + side3;
        }
    }

         class Program
        {
            static void Main(string[] args)
            {
                Square square_perimetri = new Square(4, 4);
                int s = square_perimetri.perimetri();

                Triangle triangle_perimetri = new Triangle(10, 5, 9);
                int t = triangle_perimetri.perimetri();

                Console.WriteLine("kvadratis perimetri: " + s);
                Console.WriteLine("samkutxedis perimetri: " + t);
                Console.ReadKey();
            }
        }
}
